## 🏗️ **Enterprise Architecture Highlights**

### **NgRx Signal Store Implementation**
```typescript
// Modern reactive state management with computed values
export const ExpenseStore = signalStore(
  { providedIn: 'root' },
  withState(initialState),
  withComputed(({ expenses, isOnline }) => ({
    totalExpenses: computed(() => 
      expenses().reduce((sum, exp) => sum + exp.amount, 0)
    ),
    expensesByCategory: computed(() => 
      expenses().reduce((acc, exp) => {
        acc[exp.category] = (acc[exp.category] || 0) + exp.amount;
        return acc;
      }, {} as Record<string, number>)
    ),
    canSync: computed(() => isOnline() && pendingActions().length > 0)
  })),
  withMethods((store, http = inject(HttpClient)) => ({
    // Offline-aware CRUD operations with optimistic updates
    async addExpense(expense: CreateExpenseRequest) {
      patchState(store, (state) => ({
        expenses: [...state.expenses, { ...expense, id: Date.now() }],
        pendingActions: [...state.pendingActions, { type: 'add', expense }]
      }));
      
      if (store.isOnline()) {
        await this.syncPendingActions();
      }
    }
  }))
);
```

### **Progressive Web App Service**
```typescript
// Comprehensive PWA functionality
@Injectable({ providedIn: 'root' })
export class PwaService {
  private swUpdate = inject(SwUpdate);
  private deferredPrompt: any;

  // App installation with custom prompts
  async installApp(): Promise<boolean> {
    if (this.deferredPrompt) {
      this.deferredPrompt.prompt();
      const result = await this.deferredPrompt.userChoice;
      return result.outcome === 'accepted';
    }
    return false;
  }

  // Background sync management
  private setupBackgroundSync() {
    if ('serviceWorker' in navigator && 'sync' in window.ServiceWorkerRegistration.prototype) {
      navigator.serviceWorker.ready.then(registration => {
        return registration.sync.register('background-sync');
      });
    }
  }
}
```
